11    private static String logSetting = "app.logPath";
12  
13    private static void setLogPath() {
14      // Used to set the Log4j log file location next to the entrypoint.
15      var loc = AppRelative.logDir().toString();
16      // See log4j2.xml for how we make this work.
17      System.setProperty(logSetting, loc);
18      log.trace("{}: {}", logSetting, loc);
19    }
20  
21    public static void main(String[] args) {
22      // See the developer notes.
23      System.setProperty("prism.forceGPU", "true");
24      setLogPath();
25      // The cast is needed to prevent the varargs from eating the rest of the array
26      // https://stackoverflow.com/questions/39589879/printing-an-array-with-slf4j-only-prints-the-first-element
27      log.trace("args: {}", (Object)args);
28      log.trace("sysprops: {}", System.getProperties());
29      Application.launch(GameApplication.class, args);
30    }
