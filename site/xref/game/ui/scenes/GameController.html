104   private void bindPlayer(IGridAnimator ga){
105     renderArea.setOnKeyPressed(k -> {
106       log.trace("bindplayer key pressed: {}", k);
107       switch (k.getCode()) {
108         case D -> { ga.animateUp(); } // TODO had to relabel these
109         case A -> { ga.animateDown(); }
110         case W -> { ga.animateLeft(); }
111         case S -> { ga.animateRight(); }
112 
113         // TODO event handler hacks
114         case LEFT -> { camera.setTranslateX(camera.translateXProperty().add(10).getValue()); }
115         case RIGHT -> { camera.setTranslateX(camera.translateXProperty().add(-10).getValue()); }
116         case DOWN -> { camera.setTranslateY(camera.translateYProperty().add(10).getValue()); }
117         case UP -> { camera.setTranslateY(camera.translateYProperty().add(-10).getValue()); }
118         case NUMPAD8 -> { camera.setTranslateZ(camera.getTranslateZ() + 10); }
119         case NUMPAD2 -> { camera.setTranslateZ(camera.getTranslateZ() - 10); }
120         case Q -> { System.exit(0); }
121         case R -> { model.getValue().debugFireMap(); } // Trigger world reinit
122       };
123     });
124   }
