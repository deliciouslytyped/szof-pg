41      TexturedCube n;

43      boolean inProgress;
44      int millis = 200;

46      GridAnimator(Player n) {
47        this.n = n.cube;
48        this.parent = (Group)n.getParent();
49        parent.getChildren().add(new Marker());
50      }

52      Timeline tweener(DoubleProperty prop, double offset){
53        Timeline t = new Timeline(
54                new KeyFrame(Duration.ZERO, new KeyValue(prop, prop.getValue())),
55                new KeyFrame(Duration.millis(millis), new KeyValue(prop, prop.getValue() + offset))
56        );
57        return t;
58      }
59  
60      void tryAnim(Rotate r, boolean negative, Runnable callback){
61        if (inProgress) return;
62        else inProgress = true;
63        var prop = r.angleProperty();
64        var ts = n.getTransforms();
65        ts.add(r); // TODO accumulate instead of garbage
66        var tw = tweener(prop, negative ? -90 : 90);
67        tw.setOnFinished(event -> {
68          n.getTransforms().clear();
69          n.setRotationAxis(Rotate.X_AXIS);
70          n.setRotate(0);
71          n.setRotationAxis(Rotate.Y_AXIS);
72          n.setRotate(0);
73          n.setRotationAxis(Rotate.Z_AXIS);
74          n.setRotate(0);
75          callback.run();
76          inProgress = false;
77        });
78        tw.play();
79  
80      }
81      @Override
82      public void animateUp() {
83        log.trace("animateUp");
84        var p = n.parentToLocal(0, 50, 50);
85        Rotate r = new Rotate(0, p.getX(), p.getY(), p.getZ(), Rotate.X_AXIS);
86        tryAnim(r, true, () -> {
87          parent.setTranslateZ(parent.getTranslateZ() + 100);
88          var newside = switch (n.activeSide){
89            case 1 -> 2;
90            case 2 -> 3;
91            case 3 -> 6;
92            case 4 -> 4;
93            case 5 -> 5;
94            case 6 -> 1;
95            default -> throw new IllegalStateException("Unexpected value: " + n.activeSide);
96          };
97          n.setSide(newside); // need to do it so everything is always relatiev to side on top, facing front, orientation
98        });
99      }
