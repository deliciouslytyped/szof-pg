18    // TODO: Why is this marked NonNull in the examples?
19    // Are there situations where this can be null? Does anything other than JavaFX call start?
20    // TODO: Any specific reason for marking the stage argument final in examples?
21    @Override
22    public void start(@NonNull Stage stage) throws IOException {
23      // Source: https://jenkov.com/tutorials/javafx/3d.html
24      boolean is3DSupported = Platform.isSupported(ConditionalFeature.SCENE3D);
25      if(!is3DSupported) {
26        // TODO test this
27        log.error("""
28          Sorry, 3D is not supported in JavaFX on this platform.
29          You can try checking the developer documentation for information on how to
30          attempt to forcefully enable the GPU backend with the `prism.forceGPU` property.
31          """);
32        return;
33        }
34  
35      stage.setTitle("JavaFX Game Demo");
36      // TODO: error codes using resource bundles?
37      // TODO: any good way to DRY this?
38      var scene = new Scene(FXMLLoader.load(Objects.requireNonNull(getClass().getResource("/fxml/startscreen.fxml"), "Failed to find resource `/fxml/startscreen.fxml`.")));
39      stage.setScene(scene);
40      log.info("Loaded StartScreen scene.");
41  
42      stage.show();
43    }
44  }
